390void Game::BuildRootSignature()
391{
392 CD3DX12_DESCRIPTOR_RANGE texTable;
393 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
394
395 // Root parameter can be a table, root descriptor or root constants.
396 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
397
398 // Perfomance TIP: Order from most frequent to least frequent.
399 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
400 slotRootParameter[1].InitAsConstantBufferView(0);
401 slotRootParameter[2].InitAsConstantBufferView(1);
402 slotRootParameter[3].InitAsConstantBufferView(2);
403
404 auto staticSamplers = GetStaticSamplers();
405
406 // A root signature is an array of root parameters.
407 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
408 //define an array of so - called static samplers your application can use.
409 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
410 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
411 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
412
413 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
414 ComPtr<ID3DBlob> serializedRootSig = nullptr;
415 ComPtr<ID3DBlob> errorBlob = nullptr;
416 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
417 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
418
419 if (errorBlob != nullptr)
420 {
421 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
422 }
423 ThrowIfFailed(hr);
424
425 ThrowIfFailed(md3dDevice->CreateRootSignature(
426 0,
427 serializedRootSig->GetBufferPointer(),
428 serializedRootSig->GetBufferSize(),
429 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
430}

438 //
439 D3D12_DESCRIPTOR_HEAP_DESC srvHeapDesc = {};
440 srvHeapDesc.NumDescriptors = 3;
441 srvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV;
442 srvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_SHADER_VISIBLE;
443 ThrowIfFailed(md3dDevice->CreateDescriptorHeap(&srvHeapDesc, IID_PPV_ARGS(&mSrvDescriptorHeap)));
444
445 //
446 // Fill out the heap with actual descriptors.
447 //
448 CD3DX12_CPU_DESCRIPTOR_HANDLE hDescriptor(mSrvDescriptorHeap->GetCPUDescriptorHandleForHeapStart());
449
450 auto EagleTex = mTextures["EagleTex"]->Resource;
451 auto RaptorTex = mTextures["RaptorTex"]->Resource;
452 auto DesertTex = mTextures["DesertTex"]->Resource;
453
454 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
455
456 //This mapping enables the shader resource view (SRV) to choose how memory gets routed to the 4 return components in a shader after a memory fetch.
457 //When a texture is sampled in a shader, it will return a vector of the texture data at the specified texture coordinates.
458 //This field provides a way to reorder the vector components returned when sampling the texture.
459 //D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING will not reorder the components and just return the data in the order it is stored in the texture resource.
460 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
461

463
464 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;

469

641{

643 UINT matCBByteSize = d3dUtil::CalcConstantBufferByteSize(sizeof(MaterialConstants));
644
645 auto objectCB = mCurrFrameResource->ObjectCB->Resource();
646 auto matCB = mCurrFrameResource->MaterialCB->Resource();
647
648 // For each render item...

651 auto ri = ritems[i];
652

654 cmdList->IASetIndexBuffer(&ri->Geo->IndexBufferView());
655 cmdList->IASetPrimitiveTopology(ri->PrimitiveType);
656
657 //step18
658 CD3DX12_GPU_DESCRIPTOR_HANDLE tex(mSrvDescriptorHeap->GetGPUDescriptorHandleForHeapStart());
659 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);
660
661 D3D12_GPU_VIRTUAL_ADDRESS objCBAddress = objectCB->GetGPUVirtualAddress() + ri->ObjCBIndex * objCBByteSize;
662 D3D12_GPU_VIRTUAL_ADDRESS matCBAddress = matCB->GetGPUVirtualAddress() + ri->Mat->MatCBIndex * matCBByteSize;
663
664 //step19
665 cmdList->SetGraphicsRootDescriptorTable(0, tex);
666 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
667 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
668
669 cmdList->DrawIndexedInstanced(ri->IndexCount, 1, ri->StartIndexLocation, ri->BaseVertexLocation, 0);
