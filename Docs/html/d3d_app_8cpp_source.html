72int D3DApp::Run()
73{
74 MSG msg = {0};
75
76 mTimer.Reset();
77
78 while(msg.message != WM_QUIT)
79 {
80 // If there are Window messages then process them.
81 if(PeekMessage( &msg, 0, 0, 0, PM_REMOVE ))
82 {
83 TranslateMessage( &msg );
84 DispatchMessage( &msg );
85 }
86 // Otherwise, do animation/game stuff.
87 else
88 {
89 mTimer.Tick();
90
91 if( !mAppPaused )
92 {
93 CalculateFrameStats();
94 Update(mTimer);
95 Draw(mTimer);
96 }
97 else
98 {
99 Sleep(100);
100 }
101 }
102 }
103
104 return (int)msg.wParam;
105}

141void D3DApp::OnResize()
142{
143 assert(md3dDevice);
144 assert(mSwapChain);
145 assert(mDirectCmdListAlloc);
146
147 // Flush before changing any resources.
148 FlushCommandQueue();
149
150 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
151
152 // Release the previous resources we will be recreating.
153 for (int i = 0; i < SwapChainBufferCount; ++i)
154 mSwapChainBuffer[i].Reset();
155 mDepthStencilBuffer.Reset();
156
157 // Resize the swap chain.
158 ThrowIfFailed(mSwapChain->ResizeBuffers(
159 SwapChainBufferCount,
160 mClientWidth, mClientHeight,
161 mBackBufferFormat,
162 DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH));
163
164 mCurrBackBuffer = 0;
165
166 CD3DX12_CPU_DESCRIPTOR_HANDLE rtvHeapHandle(mRtvHeap->GetCPUDescriptorHandleForHeapStart());
167 for (UINT i = 0; i < SwapChainBufferCount; i++)
168 {
169 ThrowIfFailed(mSwapChain->GetBuffer(i, IID_PPV_ARGS(&mSwapChainBuffer[i])));
170 md3dDevice->CreateRenderTargetView(mSwapChainBuffer[i].Get(), nullptr, rtvHeapHandle);
171 rtvHeapHandle.Offset(1, mRtvDescriptorSize);
172 }
173
174 // Create the depth/stencil buffer and view.
175 D3D12_RESOURCE_DESC depthStencilDesc;
176 depthStencilDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
177 depthStencilDesc.Alignment = 0;
178 depthStencilDesc.Width = mClientWidth;
179 depthStencilDesc.Height = mClientHeight;
180 depthStencilDesc.DepthOrArraySize = 1;
181 depthStencilDesc.MipLevels = 1;
182
183 // the depth buffer. Therefore, because we need to create two views to the same resource:
184 // 1. SRV format: DXGI_FORMAT_R24_UNORM_X8_TYPELESS
185 // 2. DSV Format: DXGI_FORMAT_D24_UNORM_S8_UINT
186 // we need to create the depth buffer resource with a typeless format.
187 depthStencilDesc.Format = DXGI_FORMAT_R24G8_TYPELESS;
188
189 depthStencilDesc.SampleDesc.Count = m4xMsaaState ? 4 : 1;
190 depthStencilDesc.SampleDesc.Quality = m4xMsaaState ? (m4xMsaaQuality - 1) : 0;
191 depthStencilDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
192 depthStencilDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL;
193
194 D3D12_CLEAR_VALUE optClear;
195 optClear.Format = mDepthStencilFormat;
196 optClear.DepthStencil.Depth = 1.0f;
197 optClear.DepthStencil.Stencil = 0;
198 ThrowIfFailed(md3dDevice->CreateCommittedResource(
199 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
200 D3D12_HEAP_FLAG_NONE,
201 &depthStencilDesc,
202 D3D12_RESOURCE_STATE_COMMON,
203 &optClear,
204 IID_PPV_ARGS(mDepthStencilBuffer.GetAddressOf())));
205
206 // Create descriptor to mip level 0 of entire resource using the format of the resource.
207 D3D12_DEPTH_STENCIL_VIEW_DESC dsvDesc;
208 dsvDesc.Flags = D3D12_DSV_FLAG_NONE;
209 dsvDesc.ViewDimension = D3D12_DSV_DIMENSION_TEXTURE2D;
210 dsvDesc.Format = mDepthStencilFormat;
211 dsvDesc.Texture2D.MipSlice = 0;
212 md3dDevice->CreateDepthStencilView(mDepthStencilBuffer.Get(), &dsvDesc, DepthStencilView());
213
214 // Transition the resource from its initial state to be used as a depth buffer.
215 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(mDepthStencilBuffer.Get(),
216 D3D12_RESOURCE_STATE_COMMON, D3D12_RESOURCE_STATE_DEPTH_WRITE));
217
218 // Execute the resize commands.
219 ThrowIfFailed(mCommandList->Close());
220 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
221 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
222
223 // Wait until resize is complete.
224 FlushCommandQueue();
225
226 // Update the viewport transform to cover the client area.
227 mScreenViewport.TopLeftX = 0;
228 mScreenViewport.TopLeftY = 0;
229 mScreenViewport.Width = static_cast<float>(mClientWidth);
230 mScreenViewport.Height = static_cast<float>(mClientHeight);
231 mScreenViewport.MinDepth = 0.0f;
232 mScreenViewport.MaxDepth = 1.0f;
233
234 mScissorRect = { 0, 0, mClientWidth, mClientHeight };
235}

237LRESULT D3DApp::MsgProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)
238{
239 switch( msg )
240 {
241 // WM_ACTIVATE is sent when the window is activated or deactivated.
242 // We pause the game when the window is deactivated and unpause it
243 // when it becomes active.
244 case WM_ACTIVATE:
245 if( LOWORD(wParam) == WA_INACTIVE )
246 {
247 mAppPaused = true;
248 mTimer.Stop();
249 }
250 else
251 {
252 mAppPaused = false;
253 mTimer.Start();
254 }
255 return 0;
256
257 // WM_SIZE is sent when the user resizes the window.
258 case WM_SIZE:
259 // Save the new client area dimensions.
260 mClientWidth = LOWORD(lParam);
261 mClientHeight = HIWORD(lParam);
262 if( md3dDevice )
263 {
264 if( wParam == SIZE_MINIMIZED )
265 {
266 mAppPaused = true;
267 mMinimized = true;
268 mMaximized = false;
269 }
270 else if( wParam == SIZE_MAXIMIZED )
271 {
272 mAppPaused = false;
273 mMinimized = false;
274 mMaximized = true;
275 OnResize();
276 }
277 else if( wParam == SIZE_RESTORED )
278 {
279
280 // Restoring from minimized state?
281 if( mMinimized )
282 {
283 mAppPaused = false;
284 mMinimized = false;
285 OnResize();
286 }
287
288 // Restoring from maximized state?
289 else if( mMaximized )
290 {
291 mAppPaused = false;
292 mMaximized = false;
293 OnResize();
294 }
295 else if( mResizing )
296 {
297 // If user is dragging the resize bars, we do not resize
298 // the buffers here because as the user continuously
299 // drags the resize bars, a stream of WM_SIZE messages are
300 // sent to the window, and it would be pointless (and slow)
301 // to resize for each WM_SIZE message received from dragging
302 // the resize bars. So instead, we reset after the user is
303 // done resizing the window and releases the resize bars, which
304 // sends a WM_EXITSIZEMOVE message.
305 }
306 else // API call such as SetWindowPos or mSwapChain->SetFullscreenState.
307 {
308 OnResize();
309 }
310 }
311 }
312 return 0;
313
314 // WM_EXITSIZEMOVE is sent when the user grabs the resize bars.
315 case WM_ENTERSIZEMOVE:
316 mAppPaused = true;
317 mResizing = true;
318 mTimer.Stop();
319 return 0;
320
321 // WM_EXITSIZEMOVE is sent when the user releases the resize bars.
322 // Here we reset everything based on the new window dimensions.
323 case WM_EXITSIZEMOVE:
324 mAppPaused = false;
325 mResizing = false;
326 mTimer.Start();
327 OnResize();
328 return 0;
329
330 // WM_DESTROY is sent when the window is being destroyed.
331 case WM_DESTROY:
332 PostQuitMessage(0);
333 return 0;
334
335 // The WM_MENUCHAR message is sent when a menu is active and the user presses
336 // a key that does not correspond to any mnemonic or accelerator key.
337 case WM_MENUCHAR:
338 // Don't beep when we alt-enter.
339 return MAKELRESULT(0, MNC_CLOSE);
340
341 // Catch this message so to prevent the window from becoming too small.
342 case WM_GETMINMAXINFO:
343 ((MINMAXINFO*)lParam)->ptMinTrackSize.x = 200;
344 ((MINMAXINFO*)lParam)->ptMinTrackSize.y = 200;
345 return 0;
346
347 case WM_LBUTTONDOWN:
348 case WM_MBUTTONDOWN:
349 case WM_RBUTTONDOWN:
350 OnMouseDown(wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
351 return 0;
352 case WM_LBUTTONUP:
353 case WM_MBUTTONUP:
354 case WM_RBUTTONUP:
355 OnMouseUp(wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
356 return 0;
357 case WM_MOUSEMOVE:
358 OnMouseMove(wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
359 return 0;
360 case WM_KEYUP:
361 if(wParam == VK_ESCAPE)
362 {
363 PostQuitMessage(0);
364 }
365 else if((int)wParam == VK_F2)
366 Set4xMsaaState(!m4xMsaaState);
367
368 return 0;
369 }
370
371 return DefWindowProc(hwnd, msg, wParam, lParam);
372}

374bool D3DApp::InitMainWindow()
375{
376 WNDCLASS wc;
377 wc.style = CS_HREDRAW | CS_VREDRAW;
378 wc.lpfnWndProc = MainWndProc;
379 wc.cbClsExtra = 0;
380 wc.cbWndExtra = 0;
381 wc.hInstance = mhAppInst;
382 wc.hIcon = LoadIcon(0, IDI_APPLICATION);
383 wc.hCursor = LoadCursor(0, IDC_ARROW);
384 wc.hbrBackground = (HBRUSH)GetStockObject(NULL_BRUSH);
385 wc.lpszMenuName = 0;
386 wc.lpszClassName = L"MainWnd";
387
388 if( !RegisterClass(&wc) )
389 {
390 MessageBox(0, L"RegisterClass Failed.", 0, 0);
391 return false;
392 }
393
394 // Compute window rectangle dimensions based on requested client area dimensions.
395 RECT R = { 0, 0, mClientWidth, mClientHeight };
396 AdjustWindowRect(&R, WS_OVERLAPPEDWINDOW, false);
397 int width = R.right - R.left;
398 int height = R.bottom - R.top;
399
400 mhMainWnd = CreateWindow(L"MainWnd", mMainWndCaption.c_str(),
401 WS_OVERLAPPEDWINDOW, CW_USEDEFAULT, CW_USEDEFAULT, width, height, 0, 0, mhAppInst, 0);
402 if( !mhMainWnd )
403 {
404 MessageBox(0, L"CreateWindow Failed.", 0, 0);
405 return false;
406 }
407
408 ShowWindow(mhMainWnd, SW_SHOW);
409 UpdateWindow(mhMainWnd);
410
411 return true;
412}

414bool D3DApp::InitDirect3D()
415{
416#if defined(DEBUG) || defined(_DEBUG)
417 // Enable the D3D12 debug layer.
418{
419 ComPtr<ID3D12Debug> debugController;
420 ThrowIfFailed(D3D12GetDebugInterface(IID_PPV_ARGS(&debugController)));
421 debugController->EnableDebugLayer();
422}
423#endif
424
425 ThrowIfFailed(CreateDXGIFactory1(IID_PPV_ARGS(&mdxgiFactory)));
426
427 //Hooman: I added this code to take advantage of your "stronger" GPU.
428 //In the lab, seond graphics card NVDIA is much stronger, that would stop blinking!
429 UINT i = 0;
430 IDXGIAdapter* adapter = nullptr;
431 std::vector<IDXGIAdapter*> adapterList;
432 while (mdxgiFactory->EnumAdapters(i, &adapter) != DXGI_ERROR_NOT_FOUND)
433 {
434 adapterList.push_back(adapter);
435 ++i;
436 }
437 // Try to create hardware device.
438 HRESULT hardwareResult = D3D12CreateDevice(
439 nullptr, // default adapter
440 //adapterList[1], //NVDIA adapter! at home, you can comment this out and use nullptr to point to a stronger adapter
441 D3D_FEATURE_LEVEL_12_0,
442 IID_PPV_ARGS(&md3dDevice));
443
444
445 // Fallback to WARP device.
446 if(FAILED(hardwareResult))
447 {
448 ComPtr<IDXGIAdapter> pWarpAdapter;
449 ThrowIfFailed(mdxgiFactory->EnumWarpAdapter(IID_PPV_ARGS(&pWarpAdapter)));
450
451 ThrowIfFailed(D3D12CreateDevice(
452 pWarpAdapter.Get(),
453 D3D_FEATURE_LEVEL_12_0,
454 IID_PPV_ARGS(&md3dDevice)));
455 }
456
457 ThrowIfFailed(md3dDevice->CreateFence(0, D3D12_FENCE_FLAG_NONE,
458 IID_PPV_ARGS(&mFence)));
459
460 mRtvDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
461 mDsvDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_DSV);
462 mCbvSrvUavDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
463
464 // Check 4X MSAA quality support for our back buffer format.
465 // All Direct3D 11 capable devices support 4X MSAA for all render
466 // target formats, so we only need to check quality support.
467
468 D3D12_FEATURE_DATA_MULTISAMPLE_QUALITY_LEVELS msQualityLevels;
469 msQualityLevels.Format = mBackBufferFormat;
470 msQualityLevels.SampleCount = 4;
471 msQualityLevels.Flags = D3D12_MULTISAMPLE_QUALITY_LEVELS_FLAG_NONE;
472 msQualityLevels.NumQualityLevels = 0;
473 ThrowIfFailed(md3dDevice->CheckFeatureSupport(
474 D3D12_FEATURE_MULTISAMPLE_QUALITY_LEVELS,
475 &msQualityLevels,
476 sizeof(msQualityLevels)));
477
478 m4xMsaaQuality = msQualityLevels.NumQualityLevels;
479 assert(m4xMsaaQuality > 0 && "Unexpected MSAA quality level.");
480
481#ifdef _DEBUG
482 LogAdapters();
483#endif
484
485 CreateCommandObjects();
486 CreateSwapChain();
487 CreateRtvAndDsvDescriptorHeaps();
488
489 return true;
490}

547void D3DApp::FlushCommandQueue()
548{
549 // Advance the fence value to mark commands up to this fence point.
550 mCurrentFence++;
551
552 // Add an instruction to the command queue to set a new fence point. Because we
553 // are on the GPU timeline, the new fence point won't be set until the GPU finishes
554 // processing all the commands prior to this Signal().
555 ThrowIfFailed(mCommandQueue->Signal(mFence.Get(), mCurrentFence));
556
557 // Wait until the GPU has completed commands up to this fence point.
558 if(mFence->GetCompletedValue() < mCurrentFence)
559 {
560 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
561
562 // Fire event when GPU hits current fence.
563 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrentFence, eventHandle));
564
565 // Wait until the GPU hits current fence event is fired.
566 WaitForSingleObject(eventHandle, INFINITE);
567 CloseHandle(eventHandle);
568 }
569}

591void D3DApp::CalculateFrameStats()
592{
593 // Code computes the average frames per second, and also the
594 // average time it takes to render one frame. These stats
595 // are appended to the window caption bar.
596
597 static int frameCnt = 0;
598 static float timeElapsed = 0.0f;
599
600 frameCnt++;
601
602 // Compute averages over one second period.
603 if( (mTimer.TotalTime() - timeElapsed) >= 1.0f )
604 {
605 float fps = (float)frameCnt; // fps = frameCnt / 1
606 float mspf = 1000.0f / fps;
607
608 wstring fpsStr = to_wstring(fps);
609 wstring mspfStr = to_wstring(mspf);
610
611 wstring windowText = mMainWndCaption +
612 L" fps: " + fpsStr +
613 L" mspf: " + mspfStr;
614
615 SetWindowText(mhMainWnd, windowText.c_str());
616
617 // Reset for next average.
618 frameCnt = 0;
619 timeElapsed += 1.0f;
620 }
621}
